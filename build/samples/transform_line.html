# transform_line.js

var chart = c3.generate({
    data: {
        columns: [
            ['data1', 30, 200, 100, 400, 150, 250],
            ['data2', 130, 100, 140, 200, 150, 50]
        ],
        type: 'bar'
    },
    axis: {
        x: {
            type: 'categorized'
        }
    }
});

setTimeout(function () {
    chart.toLine('data1');
}, 1000);

setTimeout(function () {
    chart.toLine('data2');
}, 2000);

setTimeout(function () {
    chart.toBar();
}, 3000);

setTimeout(function () {
    chart.toLine();
}, 4000);
